class Solution {
    class Work {
        int difficulty;
        int profit;

        Work(int difficulty, int profit) {
            this.difficulty = difficulty;
            this.profit = profit;
        }
    }

    public int maxProfitAssignment(int[] difficulty, int[] profit, int[] worker) {
        List<Work> works = new ArrayList<>();
        for (int i = 0; i < difficulty.length; i++) {
            works.add(new Work(difficulty[i], profit[i]));
        }

        Collections.sort(works, (a, b) -> a.difficulty - b.difficulty);

        int result = 0;
        for (int w : worker) {
            int max = 0;
            for (Work work : works) {
                if (work.difficulty > w) break;
                max = Math.max(max, work.profit);
            }
            result += max;
        }

        return result;
    }
}
