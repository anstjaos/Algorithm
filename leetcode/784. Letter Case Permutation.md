class Solution {
    public List<String> letterCasePermutation(String s) {
        Set<String> result = new HashSet<>();
        permutation(s, 0, result);

        return new ArrayList<>(result);
    }

    private void permutation(String cur, int index, Set<String> set) {
        if (index == cur.length()) return;

        set.add(cur);

        if ('0' <= cur.charAt(index) && cur.charAt(index) <= '9') {
            permutation(cur, index + 1, set);
        } else {
            String temp = cur.substring(0, index);
            if ('a' <= cur.charAt(index) && cur.charAt(index) <= 'z') {
                temp += (char) (cur.charAt(index) - 'a' + 'A');
            } else {
                temp += (char) (cur.charAt(index) - 'A' + 'a');
            }

            temp += cur.substring(index + 1);

            set.add(temp);
            permutation(cur, index + 1, set);
            permutation(temp, index + 1, set);
        } 
    }
}
