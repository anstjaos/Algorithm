class Solution {
    public int minSwaps(String s) {
        Stack<Character> stack = new Stack<>();

        int miss = 0;
        for (char c : s.toCharArray()) {
            if (c == '[') {
                stack.push(c);
            } else {
                if (stack.isEmpty()) {
                    miss++;
                } else {
                    stack.pop();
                }
            }
        }

        return (miss + 1) / 2;
    }
}
