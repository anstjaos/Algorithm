class Solution {
    public int countPrimes(int n) {
        if (n < 2) return 0;
        
        boolean[] isPrime = new boolean[n + 1];
        Arrays.fill(isPrime, true);

        isPrime[0] = false;
        isPrime[1] = false;

        for (int i = 2; i * i <= n; i++) {
            if (!isPrime[i]) continue;

            for (int j = i * 2; j <= n; j += i) {
                isPrime[j] = false;
            }
        }

        int result = 0;
        for (int i = 2; i < n; i++) {
            if (isPrime[i]) result++;
        }

        return result;
    }
}
