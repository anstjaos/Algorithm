class Solution {
    public String trimTrailingVowels(String s) {
        int lastIndex = s.length() - 1;

        while (lastIndex >= 0 && isVowel(s.charAt(lastIndex))) {
            lastIndex--;
        }

        return s.substring(0, lastIndex + 1);
    }

    private boolean isVowel(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
}
