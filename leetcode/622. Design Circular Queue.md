class MyCircularQueue {
    int[] dequeue;
    int size;
    int front = 0;
    int rear = -1;

    public MyCircularQueue(int k) {
        size = k;
        dequeue = new int[size];
    }
    
    public boolean enQueue(int value) {
        if (isFull()) return false;
        rear = (rear + 1) % size;
        dequeue[rear] = value;
        return true;
    }
    
    public boolean deQueue() {
        if (isEmpty()) return false;
        if (front == rear) {
            front = 0;
            rear = -1;
        } else {
            front = (front + 1) % size;
        }

        return true;
    }
    
    public int Front() {
        if (isEmpty()) return -1;
        return dequeue[front];
    }
    
    public int Rear() {
        if (isEmpty()) return -1;
        return dequeue[rear];
    }
    
    public boolean isEmpty() {
        return rear == -1;
    }
    
    public boolean isFull() {
        return !isEmpty() && (rear + 1) % size == front;
    }
}
