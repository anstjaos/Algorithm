class Solution {
    class Temperature {
        int day, temp;

        Temperature(int day, int temp) {
            this.day = day;
            this.temp = temp;
        }
    }

    public int[] dailyTemperatures(int[] temperatures) {
        int[] result = new int[temperatures.length];
        Stack<Temperature> stack = new Stack<>();

        for (int i = 0; i < temperatures.length; i++) {
            int temp = temperatures[i];
            while (!stack.isEmpty() && stack.peek().temp < temp) {
                Temperature top = stack.pop();
                result[top.day] = i - top.day;
            }
            stack.push(new Temperature(i, temp));
        }

        return result;
    }
}
