class Solution {
    class Trie {
        Trie[] dict;
        boolean isEnd;

        Trie() {
            this.dict = new Trie[26];
            this.isEnd = false;
        }
    }

    public String replaceWords(List<String> dictionary, String sentence) {
        Trie trie = new Trie();
        for (String word : dictionary) {
            Trie cur = trie;
            for (char c : word.toCharArray()) {
                int index = c - 'a';
                if (cur.dict[index] == null) {
                    cur.dict[index] = new Trie();
                }
                cur = cur.dict[index];
            }

            cur.isEnd = true;
        }

        StringBuilder sb = new StringBuilder();
        String[] words = sentence.split(" ");

        for (int i = 0; i < words.length; i++) {
            String word = changeWord(words[i], trie);
            sb.append(word);
            if (i != words.length - 1) {
                sb.append(" ");
            }
        }

        return sb.toString();
    }

    private String changeWord(String word, Trie trie) {
        String result = "";
        Trie cur = trie;

        for (int i = 0; i < word.length(); i++) {
            int index = word.charAt(i) - 'a';
            if (cur.dict[index] == null) {
                return word;
            }

            cur = cur.dict[index];
            result += word.charAt(i);
            if (cur.isEnd) {
                return result;
            }
        }

        return word;
    }
}
