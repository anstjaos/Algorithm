class Solution {
    public int minMoves(int[] nums, int limit) {
        int length = nums.length;
        int[] delta = new int[limit * 2 + 2];

        for (int i = 0; i < length / 2; i++) {
            int min = Math.min(nums[i], nums[length - i - 1]);
            int max = Math.max(nums[i], nums[length - i - 1]);

            delta[2] += 2;
            delta[min + 1]--;
            delta[min + max]--;
            delta[min + max + 1]++;
            delta[max + limit + 1]++;
        }

        int result = length, moves = 0;
        for (int i = 2; i <= limit * 2; i++) {
            moves += delta[i];
            result = Math.min(result, moves);
        }

        return result;
    }
}
