class Solution {
    class Water {
        int x, y;

        Water(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    public boolean canMeasureWater(int x, int y, int target) {
        if (x + y == target) return true;
        if (x + y < target) return false;
        if (x % 2 == 0 && y % 2 == 0 && target % 2 != 0) return false;

        Queue<Water> queue = new LinkedList<>();
        queue.add(new Water(0, 0));
        Set<String> visit = new HashSet<>();

        while (!queue.isEmpty()) {
            Water cur = queue.poll();
            if (cur.x + cur.y == target) return true;
            visit.add(cur.x + "," + cur.y);

            int nextX, nextY;

            nextX = cur.x - Math.min(cur.x, y - cur.y);
            nextY = cur.y + Math.min(cur.x, y - cur.y);
            if (!visit.contains(nextX + "," + nextY)) {
                queue.add(new Water(nextX, nextY));
            }

            nextX = cur.x + Math.min(x - cur.x, cur.y);
            nextY = cur.y - Math.min(x - cur.x, cur.y);
            if (!visit.contains(nextX + "," + nextY)) {
                queue.add(new Water(nextX, nextY));
            }

            nextX = 0;
            nextY = cur.y;
            if (!visit.contains(nextX + "," + nextY)) {
                queue.add(new Water(nextX, nextY));
            }

            nextX = cur.x;
            nextY = 0;
            if (!visit.contains(nextX + "," + nextY)) {
                queue.add(new Water(nextX, nextY));
            }

            nextX = x;
            nextY = cur.y;
            if (!visit.contains(nextX + "," + nextY)) {
                queue.add(new Water(nextX, nextY));
            }

            nextX = cur.x;
            nextY = y;
            if (!visit.contains(nextX + "," + nextY)) {
                queue.add(new Water(nextX, nextY));
            }
        }

        return false;
    }
}
