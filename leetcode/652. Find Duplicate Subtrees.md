/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<TreeNode> findDuplicateSubtrees(TreeNode root) {
        Map<String, Integer> nodeMap = new HashMap<>();
        List<TreeNode> result = new ArrayList<>();

        findDuplicate(result, root, nodeMap);

        return result;
    }

    private String findDuplicate(List<TreeNode> result, TreeNode root, Map<String, Integer> map) {
        if (root == null) return "";

        String left = findDuplicate(result, root.left, map);
        String right = findDuplicate(result, root.right, map);

        String str = root.val + "," + left + "," + right;
        if (map.getOrDefault(str, 0) == 1) {
            result.add(root);
        }

        map.put(str, map.getOrDefault(str, 0) + 1);
        return str;
    }
}
