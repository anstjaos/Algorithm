class Solution {
    public String reverseByType(String s) {
        Stack<Character> lowerStack = new Stack<>();
        Stack<Character> specialStack = new Stack<>();

        for (char c : s.toCharArray()) {
            if ('a' <= c && c <= 'z') {
                lowerStack.add(c);
            } else {
                specialStack.add(c);
            }
        }

        StringBuilder sb = new StringBuilder();
        for (char c : s.toCharArray()) {
            if ('a' <= c && c <= 'z') {
                sb.append(lowerStack.pop());
            } else {
                sb.append(specialStack.pop());
            }
        }

        return sb.toString();
    }
}
