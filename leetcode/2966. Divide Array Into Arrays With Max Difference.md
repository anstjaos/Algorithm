class Solution {
    public int[][] divideArray(int[] nums, int k) {
        int[][] result = new int[nums.length / 3][3];
        Arrays.sort(nums);
        for (int i = 0; i * 3 < nums.length; i++) {
            for (int j = 0; j < 3; j++) {
                result[i][j] = nums[i * 3 + j];
            }

            if (result[i][2] - result[i][0] > k) {
                return new int[0][0];
            }
        }

        return result;
    }
}
