class Solution {
    class Point {
        int x, y;
        
        Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    public int maxAreaOfIsland(int[][] grid) {
        int result = 0;
        boolean[][] isVisit = new boolean[grid.length][grid[0].length];

        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0 || isVisit[i][j]) continue;
                result = Math.max(result, getAreaSize(grid, i, j, isVisit));
            }
        }

        return result;
    }

    private int getAreaSize(int[][] grid, int x, int y, boolean[][] isVisit) {
        int[] dirX = {-1, 0, 1, 0}, dirY = {0, 1, 0, -1};
        int size = 0;

        Queue<Point> queue = new LinkedList<>();
        isVisit[x][y] = true;
        queue.add(new Point(x, y));

        while (!queue.isEmpty()) {
            Point cur = queue.poll();
            size++;

            for (int i = 0; i < 4; i++) {
                int nextX = cur.x + dirX[i];
                int nextY = cur.y + dirY[i];

                if (nextX < 0 || nextY < 0 || nextX >= grid.length || nextY >= grid[0].length) {
                    continue;
                }
                if (grid[nextX][nextY] == 0 || isVisit[nextX][nextY]) continue;

                isVisit[nextX][nextY] = true;
                queue.add(new Point(nextX, nextY));
            }
        }

        return size;
    }
}
