class Solution {
    public int[] deckRevealedIncreasing(int[] deck) {
        Arrays.sort(deck);

        int length = deck.length;
        int[] result = new int[length];
        Deque<Integer> deque = new LinkedList<>();

        for (int i = 0; i < length; i++) {
            deque.add(i);
        }

        for (int card : deck) {
            int index = deque.poll();
            result[index] = card;
            if (!deque.isEmpty()) {
                deque.add(deque.poll());
            }
        }

        return result;
    }
}
