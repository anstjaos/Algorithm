class Solution {
    public boolean circularArrayLoop(int[] nums) {
        boolean[] isVisit = new boolean[nums.length];

        for (int i = 0; i < nums.length; i++) {
            if (isVisit[i]) continue;
            int cur = i;
            int next = i;
            boolean isForward = nums[i] >= 0;

            while (true) {
                cur = findNextIndex(nums, isForward, cur);
                if (cur == -1) break;

                next = findNextIndex(nums, isForward, next);
                if (next == -1) break;
                next = findNextIndex(nums, isForward, next);
                if (next == -1) break;

                if (cur == next) return true;

                isVisit[cur] = true;
                isVisit[next] = true;
            }
        }

        return false;
    }

    private int findNextIndex(int[] arr, boolean isForward, int cur) {
        boolean direction = arr[cur] >= 0;
        if (direction != isForward) {
            return -1;
        }
        int nextIndex = (arr[cur] + cur) % arr.length;
        if (nextIndex < 0) {
            nextIndex += arr.length;
        }
        if (nextIndex == cur) {
            return -1;
        }
        return nextIndex;
    }
}
