class Solution {
    public boolean isValid(String s) {
        Stack<Character> stack = new Stack<>();

        for (char c : s.toCharArray()) {
            if (c == 'c') {
                if (stack.size() < 2) return false;
                if (stack.peek() != 'b') return false;

                stack.pop();
                if (stack.peek() != 'a') return false;
                stack.pop();
                continue;
            }

            stack.push(c);
        }

        return stack.isEmpty();
    }
}
