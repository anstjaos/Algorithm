class Solution {
    public int numberOfBoomerangs(int[][] points) {
        int length = points.length;
        if (length < 3) return 0;
        int result = 0;

        for (int i = 0; i < length; i++) {
            Map<Long, Integer> map = new HashMap<>();
            for (int j = 0; j < length; j++) {
                if (i == j) continue;

                long distance = getDistance(points[i], points[j]);
                map.put(distance, map.getOrDefault(distance, 0) + 1);
            }

            for (Map.Entry<Long, Integer> cur : map.entrySet()) {
                int val = cur.getValue();
                if (val> 1) result += val * (val - 1);
            }
        }

        return result;
    }

    private long getDistance(int[] point1, int[] point2) {
        int x = point1[0] - point2[0];
        int y = point1[1] - point2[1];

        return y * y + x * x;
    }
}
