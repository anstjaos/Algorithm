class Solution {
    public int[] scoreValidator(String[] events) {
        int[] result = new int[2];
        for (String event : events) {
            if (event.equals("W")) {
                result[1]++;
            } else if (event.equals("WD")) {
                result[0]++;
            } else if (event.equals("NB")) {
                result[0]++;
            } else {
                int num = Integer.parseInt(event);
                result[0] += num;
            }

            if (result[1] == 10) break;
        }

        return result;
    }
}
