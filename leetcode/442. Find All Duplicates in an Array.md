class Solution {
    public List<Integer> findDuplicates(int[] nums) {
        List<Integer> result = new ArrayList<>();
        
        boolean[] isExists = new boolean[100006];

        for (int num : nums) {
            if (isExists[num]) {
                result.add(num);
            }

            isExists[num] = true;
        }

        return result;
    }
}
