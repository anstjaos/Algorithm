class Solution {
    public int minCost(int n, int[][] edges) {
        List<List<int[]>> out = new ArrayList<>();
        List<List<int[]>> in = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            out.add(new ArrayList<>());
            in.add(new ArrayList<>());
        }

        for (int[] edge : edges) {
            out.get(edge[0]).add(new int[]{edge[1], edge[2]});
            in.get(edge[1]).add(new int[]{edge[0], edge[2]});
        }

        long INF = (long) 1e18;
        long[][] dist = new long[n][2];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dist[i], INF);
        }

        PriorityQueue<long[]> pq = new PriorityQueue<>((a, b) -> Long.compare(a[0], b[0]));

        dist[0][0] = 0;
        pq.add(new long[]{0, 0, 0});

        while (!pq.isEmpty()) {
            long[] cur = pq.poll();
            long cost = cur[0];
            int node = (int) cur[1];
            int used = (int) cur[2];

            if (cost > dist[node][used]) continue;

            for (int[] edge : out.get(node)) {
                int nextNode = edge[0];
                int nextCost = edge[1];

                if (dist[nextNode][0] > cost + nextCost) {
                    dist[nextNode][0] = cost + nextCost;
                    pq.add(new long[]{dist[nextNode][0], nextNode, 0});
                }
            }

            if (used == 0) {
                for (int[] edge : in.get(node)) {
                    int nextNode = edge[0];
                    int nextCost = edge[1];

                    if (dist[nextNode][0] > cost + 2L * nextCost) {
                        dist[nextNode][0] = cost + 2L * nextCost;
                        pq.add(new long[]{dist[nextNode][0], nextNode, 0});
                    }
                }
            }
        }

        long result = Math.min(dist[n - 1][0], dist[n - 1][1]);
        return result >= INF ? -1 : (int) result;
    }
}
