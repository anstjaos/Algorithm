/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int left = 0;
    int maxDepth = 0;

    public int findBottomLeftValue(TreeNode root) {
        if (root.left == null && root.right == null) return root.val;

        left = root.val;
        getLeftValue(root, 0);

        return left;
    }

    private void getLeftValue(TreeNode cur, int depth) {
        if (cur.left != null) {
            if (depth + 1 > maxDepth) {
                left = cur.left.val;
                maxDepth = depth + 1;
            }

            getLeftValue(cur.left, depth + 1);
        }

        if (cur.right != null) {
            if (depth + 1 > maxDepth) {
                left = cur.right.val;
                maxDepth = depth + 1;
            }
            getLeftValue(cur.right, depth + 1);
        }
    }
}
