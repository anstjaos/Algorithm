class Solution {
    public int maxCoins(int[] piles) {
        int max = 0;
        for (int pile : piles) {
            max = Math.max(max, pile);
        }

        int[] freq = new int[max + 1];
        for (int pile : piles) {
            freq[pile]++;
        }

        int coins = 0, chance = piles.length / 3;

        int turn = 1;
        int cur = max;

        while (chance != 0) {
            if (freq[cur] > 0) {
                if (turn == 1) turn = 0;
                else {
                    chance--;
                    turn = 1;
                    coins += cur;
                }

                freq[cur]--;
            } else {
                cur--;
            }
        }

        return coins;
    }
}
