class Solution {
    public int wiggleMaxLength(int[] nums) {
        if (nums.length <= 1) return nums.length;

        int idx = 0;
		while (idx < nums.length - 1 && nums[idx] == nums[idx + 1]) {
			idx++;
		}

        if (idx == nums.length - 1) return 1;

        boolean isPositive = (nums[idx + 1] - nums[idx]) > 0;
        int result = 2;

        for (int i = idx + 1; i < nums.length - 1; i++) {
			if (isPositive && nums[i + 1] < nums[i]) {
				nums[result] = nums[i + 1];
				result++;
				isPositive = !isPositive;
			} else {
				if (!isPositive && nums[i + 1] > nums[i]) {
					nums[result] = nums[i + 1];
					result++;
					isPositive = !isPositive;
				}
			}
		}

        return result;
    }
}
