class Solution {
    public String largestEven(String s) {
        int lastIndex = s.length() - 1;
        for (; lastIndex >= 0; lastIndex--) {
            if (s.charAt(lastIndex) == '2') break;
        }

        if (lastIndex == -1) return "";

        return s.substring(0, lastIndex + 1);
    }
}
