class Solution {
    public boolean checkGoodInteger(int n) {
        int digitSum = 0, squareSum = 0;
        while (n != 0) {
            int num = n % 10;
            n /= 10;

            digitSum += num;
            squareSum += (num * num);
        }

        return squareSum - digitSum >= 50;
    }
}
