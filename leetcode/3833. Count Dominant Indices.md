class Solution {
    public int dominantIndices(int[] nums) {
        long sum = 0;
        int result = 0;

        for (int i = nums.length - 1; i > 0; i--) {
            sum += nums[i];
            if (nums[i - 1] > sum / (nums.length - i)) {
                result++;
            }
        }

        return result;
    }
}
