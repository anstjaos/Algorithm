class Solution {
    public int findRadius(int[] houses, int[] heaters) {
        if (houses == null || houses.length == 0 || heaters == null || heaters.length == 0) {
            return 0;
        }

        Arrays.sort(houses);
        Arrays.sort(heaters);

        int houseLength = houses.length;
        int heaterLength = heaters.length;

        int houseIndex = 0, heaterIndex = 0;
        int result = 0;

        while (houseIndex < houseLength && heaterIndex < heaterLength) {
            int dist1 = Math.abs(heaters[heaterIndex] - houses[houseIndex]);
            int dist2 = Integer.MAX_VALUE;

            if (heaterIndex + 1 < heaterLength) {
                dist2 = Math.abs(heaters[heaterIndex + 1] - houses[houseIndex]);
            }

            if (dist1 < dist2) {
                result = Math.max(result, dist1);
                houseIndex++;
            } else {
                heaterIndex++;
            }
        }

        return result;
    }
}
