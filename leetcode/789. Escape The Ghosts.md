class Solution {
    public boolean escapeGhosts(int[][] ghosts, int[] target) {
        int length = Math.abs(target[0]) + Math.abs(target[1]);

        for (int[] ghost : ghosts) {
            if (length >= Math.abs(target[0] - ghost[0]) + Math.abs(target[1] - ghost[1])) {
                return false;
            }
        }

        return true;
    }
}
