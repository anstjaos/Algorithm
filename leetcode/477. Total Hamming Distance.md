class Solution {
    public int totalHammingDistance(int[] nums) {
        int result = 0;
        int length = nums.length;

        for (int j = 0; j < 32; j++) {
            int bitCount = 0;
            for (int i = 0; i < length; i++) {
                bitCount += (nums[i] >> j) & 1;
            }
            result += bitCount * (length - bitCount);
        }

        return result;
    }
}
