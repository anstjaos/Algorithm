class Solution {
    public int smallestAbsent(int[] nums) {
        int average = 0;
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            average += num;
            set.add(num);
        }

        average = average / nums.length + 1;
        if (average <= 0) {
            average = 1;
        }

        while (set.contains(average)) {
            average++;
        }

        return average;
    }
}
