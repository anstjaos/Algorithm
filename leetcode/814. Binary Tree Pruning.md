/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode pruneTree(TreeNode root) {
        boolean result = removeZeroTree(root);
        if (result) {
            return null;
        }
        
        return root;
    }

    private boolean removeZeroTree(TreeNode root) {
        if (root == null) return true;

        boolean left = removeZeroTree(root.left);
        boolean right = removeZeroTree(root.right);

        if (left) {
            root.left = null;
        }
        if (right) {
            root.right = null;
        }

        return left && right && root.val == 0;
    }
}
