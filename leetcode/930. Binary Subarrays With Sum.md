class Solution {
    public int numSubarraysWithSum(int[] nums, int goal) {
        return solve(nums, goal) - solve(nums, goal - 1);
    }

    private int solve(int[] nums, int goal) {
        int left = 0, right = 0;

        int result = 0;
        int cur = 0;

        for (; right < nums.length; right++) {
            cur += nums[right];
            while (cur > goal && left <= right) {
                cur -= nums[left];
                left++;
            }

            result += right - left + 1;
        }

        return result;
    }
}
