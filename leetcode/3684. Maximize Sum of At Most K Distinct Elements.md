class Solution {
    public int[] maxKDistinct(int[] nums, int k) {
        int[] result = new int[k];
        int index = 0;
        Set<Integer> visit = new HashSet<>();

        Arrays.sort(nums);
        for (int i = nums.length - 1; i >= 0; i--) {
            if (visit.contains(nums[i])) continue;
            result[index++] = nums[i];
            visit.add(nums[i]);
            if (index == k) break;
        }
        if (index != k) {
            int[] temp = result;
            result = new int[index];
            for (int i = 0; i < index; i++) {
                result[i] = temp[i];
            }
        }
        return result;
    }
}
