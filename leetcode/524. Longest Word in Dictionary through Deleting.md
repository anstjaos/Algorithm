class Solution {
    public String findLongestWord(String s, List<String> dictionary) {
        String result = "";

        for (String word : dictionary) {
            int wordLength = word.length(), resultLength = result.length();
            if (wordLength < resultLength 
                || (wordLength == resultLength && word.compareTo(result) > 0)) continue;

            int pos = -1;
            for (int i = 0; i < wordLength; i++) {
                pos = s.indexOf(word.charAt(i), pos + 1);
                if (pos == -1) break;
            }
            if (pos != -1) result = word;
        }

        return result;
    }
}
