class Solution {
    public int findPaths(int m, int n, int maxMove, int startRow, int startColumn) {
        final int MODULO = 1000000007;
        int[][] dp = new int[m][n];
        dp[startRow][startColumn] = 1;
        int count = 0;

        for (int move = 0; move < maxMove; move++) {
            int[][] temp = new int[m][n];

            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    if (i == m - 1) count = (count + dp[i][j]) % MODULO;
                    if (j == n - 1) count = (count + dp[i][j]) % MODULO;
                    if (i == 0) count = (count + dp[i][j]) % MODULO;
                    if (j == 0) count = (count + dp[i][j]) % MODULO;

                    temp[i][j] = (
                        ((i > 0 ? dp[i - 1][j] : 0) + (i < m - 1 ? dp[i + 1][j] : 0)) % MODULO
                        + ((j > 0 ? dp[i][j - 1] : 0) + (j < n - 1 ? dp[i][j + 1] : 0)) % MODULO
                    ) % MODULO;
                }
            }

            dp = temp;
        }

        return count;
    }
}
