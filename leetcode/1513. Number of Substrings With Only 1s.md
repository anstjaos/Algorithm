class Solution {
    public int numSub(String s) {
        int mod = 1000000007;
        long count = 0, result = 0;

        for (char c : s.toCharArray()) {
            result += (1 - (c - '0')) * count * (count + 1) / 2;
            count = (c - '0') * (count + 1);
        }

        result += count * (count + 1) / 2;
        return (int) (result % mod);
    }
}
