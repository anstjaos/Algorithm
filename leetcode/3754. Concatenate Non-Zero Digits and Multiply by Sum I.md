class Solution {
    public long sumAndMultiply(int n) {
        long sum = 0;
        StringBuilder sb = new StringBuilder();

        while (n != 0) {
            int num = n % 10;
            n /= 10;
            if (num == 0) continue;

            sb.append(num);
            sum += num;
        }
        if (sb.length() == 0) return 0;

        long num = Long.parseLong(sb.reverse().toString());
        return num * sum;
    }
}
