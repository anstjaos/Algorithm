class Solution {
    public String minWindow(String s, String t) {
        if (s.length() < t.length()) {
            return "";
        }

        int[] tCount = new int[129];
        for (Character c : t.toCharArray()) {
            tCount[c]++;
        }

        int left = 0, right = 0, minLen = Integer.MAX_VALUE;
        int count = t.length();
        char[] sIndex = s.toCharArray();
        int startIndex = 0;

        while (right < s.length()) {
            if (tCount[sIndex[right++]]-- > 0) {
                count--;
            }

            while (count == 0) {
                if (right - left < minLen) {
                    startIndex = left;
                    minLen = right - left;
                }

                if (tCount[sIndex[left++]]++ == 0) {
                    count++;
                }
            }
        }

        return minLen == Integer.MAX_VALUE ? "" : new String(sIndex, startIndex, minLen);
    }
}

class Solution {
    public String minWindow(String s, String t) {
        int left = 0, right = t.length();
        int minLength = Integer.MAX_VALUE;
        String result = "";

        for (; right <= s.length(); right++) {
            String substr = s.substring(left, right);

            while (isPossible(substr, t)) {
                if (right - left < minLength) {
                    result = substr;
                    minLength = right - left;
                }

                left++;
                substr = s.substring(left, right);
            }
        }

        return result;
    }

    private boolean isPossible(String s, String t) {
        int[] sCount = new int[256];
        for (char c : s.toCharArray()) {
            sCount[c]++;
        }

        for (char c : t.toCharArray()) {
            sCount[c]--;
            if (sCount[c] < 0) return false;
        }

        return true;
    }
}
