class Solution {
    public int gcdOfOddEvenSums(int n) {
        int odd = 0;
        for (int i = 0; i < n; i++) {
            odd += i * 2 + 1;
        }

        int even = 0;
        for (int i = 1; i <= n; i++) {
            even += i * 2;
        }

        return gcd(odd, even);
    }

    private int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }

        return a;
    }
}
