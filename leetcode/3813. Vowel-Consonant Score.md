class Solution {
    public int vowelConsonantScore(String s) {
        int v = 0, cCount = 0;

        for (char c : s.toCharArray()) {
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') v++;
            else if ('a' <= c && c <= 'z') cCount++;
        }

        return cCount > 0 ? (int) Math.floor(v / cCount) : 0;
    }
}
