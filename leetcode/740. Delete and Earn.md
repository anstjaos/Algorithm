class Solution {
    public int deleteAndEarn(int[] nums) {
        int[] values = new int[10001];
        for (int num : nums) {
            values[num] += num;
        }

        int take = 0, skip = 0;
        for (int i = 0; i < 10001; i++) {
            int cur = skip + values[i];
            int sk = Math.max(skip, take);
            take = cur;
            skip = sk;
        }

        return Math.max(take, skip);
    }
}
