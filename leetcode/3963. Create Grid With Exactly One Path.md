class Solution {
    public String[] createGrid(int m, int n) {
        char[][] grid = new char[m][n];
        for (int i = 0; i < m; i++) {
            Arrays.fill(grid[i], '.');
        }

        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n - 1; j++) {
                grid[i][j] = '#';
            }
        }

        String[] result = new String[m];
        for (int i = 0; i < m; i++) {
            String temp = "";
            for (int j = 0; j < n; j++) {
                temp += grid[i][j];
            }

            result[i] = temp;
        }

        return result;
    }
}
