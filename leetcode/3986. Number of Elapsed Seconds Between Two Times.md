class Solution {
    public int secondsBetweenTimes(String startTime, String endTime) {
        String[] parsedStart = startTime.split(":");
        String[] parsedEnd = endTime.split(":");

        int startSecond = getSecond(parsedStart);
        int endSecond = getSecond(parsedEnd);

        return endSecond - startSecond;
    }

    private int getSecond(String[] parsed) {
        return Integer.parseInt(parsed[0]) * 3600 + Integer.parseInt(parsed[1]) * 60 + Integer.parseInt(parsed[2]);
    }
}
