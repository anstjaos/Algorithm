class Solution {
    public boolean checkStrings(String s1, String s2) {
        int[] count = new int[52];

        for (int i = 0; i < s1.length(); i++) {
            int offset = (i & 1) * 26;
            count[s1.charAt(i) - 'a' + offset]++;
            count[s2.charAt(i) - 'a' + offset]--;
        }

        for (int i = 0; i < 52; i++) {
            if (count[i] != 0) return false;
        }

        return true;
    }
}
