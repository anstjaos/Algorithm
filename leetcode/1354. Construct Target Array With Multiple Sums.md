class Solution {
    public boolean isPossible(int[] target) {
        if (target.length == 1) {
            return target[0] == 1;
        }

        long sum = 0;
        PriorityQueue<Long> pq = new PriorityQueue<>(Collections.reverseOrder());
        for (int num : target) {
            pq.add((long) num);
            sum += num;
        }

        while (pq.peek() != 1) {
            long cur = pq.poll();

            if (sum - cur == 1) return true;

            long x = cur % (sum - cur);
            sum = sum - cur + x;

            if (x == 0 || x == cur) return false;
            pq.add(x);
        }

        return true;
    }
}
