class Solution {
    int result = 0;
    public int countArrangement(int n) {
        if (n == 0) return 0;
        getCount(1, n , new boolean[n + 1]);
        return result;
    }

    private void getCount(int index, int n, boolean[] isUsed) {
        if (index > n) {
            result++;
            return;
        }

        for (int i = 1; i <= n; i++) {
            if (!isUsed[i] && (index % i == 0 || i % index == 0)) {
                isUsed[i] = true;
                getCount(index + 1, n, isUsed);
                isUsed[i] = false;
            }
        }
    }
}
