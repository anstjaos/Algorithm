class Solution {
    public int[] limitOccurrences(int[] nums, int k) {
        int[] counts = new int[105];
        int length = 0;

        for (int num : nums) {
            if (counts[num] == k) continue;

            counts[num]++;
            length++;
        }

        int[] result = new int[length];
        int index = 0;
        for (int i = 1; i < 105; i++) {
            while (counts[i] > 0) {
                result[index++] = i;
                counts[i]--;
            }
        }

        return result;
    }
}
