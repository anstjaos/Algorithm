/*
// Definition for a Node.
class Node {
    public int val;
    public Node prev;
    public Node next;
    public Node child;
};
*/

class Solution {
    public Node flatten(Node head) {
        Node cur = head;
        Node next = cur;
        Stack<Node> stack = new Stack<>();

        while (cur != null) {
            if (cur.child != null) {
                if (cur.next != null) {
                    stack.add(cur.next);
                }
                
                cur.child.prev = cur;
                cur.next = cur.child;
                next = cur.child;
                cur.child = null;
            } else {
                next = cur.next;
            }

            if (next == null && !stack.isEmpty()) {
                next = stack.pop();
                next.prev = cur;
                cur.next = next;
            }

            cur = next;
        }

        return head;
    }
}
