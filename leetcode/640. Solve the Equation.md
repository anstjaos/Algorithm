class Solution {
    public String solveEquation(String equation) {
        String[] parsed = equation.split("=");
        int leftXCount = 0, leftSum = 0;
        
        boolean isPlus = true;
        int num = -1;
        for (char c : parsed[0].toCharArray()) {
            if (c == '+') {
                if (num != -1) {
                    if (isPlus) leftSum += num;
                    else leftSum -= num;
                }

                isPlus = true;
                num = -1;
            } else if (c == '-') {
                if (num != -1) {
                    if (isPlus) leftSum += num;
                    else leftSum -= num;
                }

                isPlus = false;
                num = -1;
            } else if (c == 'x') {
                if (num == -1) num = 1;
                
                if (isPlus) leftXCount += num;
                else leftXCount -= num;

                num = -1;
            } else {
                if (num == -1) num = 0;

                num *= 10;
                num += (int) (c - '0');
            }
        }

        if (num != -1) {
            if (isPlus) leftSum += num;
            else leftSum -= num;
        }
        

        int rightXCount = 0, rightSum = 0;
        isPlus = true;
        num = -1;
        for (char c : parsed[1].toCharArray()) {
            if (c == '+') {
                if (num != -1) {
                    if (isPlus) rightSum += num;
                    else rightSum -= num;
                }
                
                isPlus = true;
                num = -1;
            } else if (c == '-') {
                if (num != -1) {
                    if (isPlus) rightSum += num;
                    else rightSum -= num;
                }

                isPlus = false;
                num = -1;
            } else if (c == 'x') {
                if (num == -1) num = 1;
                
                if (isPlus) rightXCount += num;
                else rightXCount -= num;

                num = -1;
            } else {
                if (num == -1) num = 0;

                num *= 10;
                num += (int) (c - '0');
            }
        }

        if (num != -1) {
            if (isPlus) rightSum += num;
            else rightSum -= num;
        }
       
        int xCount = leftXCount - rightXCount;
        int sum = rightSum - leftSum;

        if (xCount == 0) {
            if (sum == 0) {
                return "Infinite solutions";
            }
            return "No solution";
        }

        int result = sum / xCount;
        return "x=" + result;
    }
}
