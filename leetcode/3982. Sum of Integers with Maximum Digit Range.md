class Solution {
    public int maxDigitRange(int[] nums) {
        List<Integer> maxNums = new ArrayList<>();
        int maxDigitRange = 0;

        for (int num : nums) {
            int max = 0, min = 10;
            int temp = num;
            while (temp != 0) {
                max = Math.max(max, temp % 10);
                min = Math.min(min, temp % 10);

                temp /= 10;
            }

            int diff = max - min;

            if (diff == maxDigitRange) {
                maxNums.add(num);
            } else if (diff > maxDigitRange) {
                maxDigitRange = diff;
                maxNums = new ArrayList<>();
                maxNums.add(num);
            }
        }

        int result = 0;
        for (int num : maxNums) {
            result += num;
        }

        return result;
    }
}
