class Solution {
    public int totalWaviness(int num1, int num2) {
        int count = 0;

        for (int i = num1; i <= num2; i++) {
            if (i < 100) continue;
            count += getWaivness(String.valueOf(i));
        }

        return count;
    }

    private int getWaivness(String num) {
        int count = 0;
        for (int i = 0; i <= num.length() - 3; i++) {
            String temp = num.substring(i, i + 3);

            if (temp.charAt(0) < temp.charAt(1) && temp.charAt(1) > temp.charAt(2)) {
                count++;
            } else if (temp.charAt(0) > temp.charAt(1) && temp.charAt(1) < temp.charAt(2)) {
                count++;
            }
        }

        return count;
    }
}
