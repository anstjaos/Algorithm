class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int[] wordsCount = new int[11];
        for (int i = 0; i < words.length; i++) {
            int count = getMinFrequency(words[i]);
            wordsCount[count]++;
        }

        int sum = 0;
        for (int i = 0; i < wordsCount.length; i++) {
            sum += wordsCount[i];
            wordsCount[i] = sum;
        }

        int[] result = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            String query = queries[i];
            int frequency = getMinFrequency(query);
            
            result[i] = wordsCount[wordsCount.length - 1] - wordsCount[frequency];
        }

        return result;
    }
    
    private int getMinFrequency(String word) {
        int[] count = new int[26];
        for (char c : word.toCharArray()) {
            count[c - 'a']++;
        }

        int result = 0;
        for (int i = 0; i < 26; i++) {
            if (count[i] != 0) {
                result = count[i];
                break;
            }
        }

        return result;
    }
}
