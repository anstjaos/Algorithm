class Solution {
    class Trie {
        Trie[] trie;
        boolean isEnd;

        Trie() {
            trie = new Trie[10];
            isEnd = false;
        }
    }

    public int longestCommonPrefix(int[] arr1, int[] arr2) {
        Trie root = new Trie();
        for (int num : arr1) {
            insert(root, String.valueOf(num));
        }

        int result = 0;
        for (int num : arr2) {
            result = Math.max(result, getLongestPrefix(root, String.valueOf(num)));
        }

        return result;
    }

    private void insert(Trie root, String num) {
        Trie cur = root;
        for (char c : num.toCharArray()) {
            int index = c - '0';

            if (cur.trie[index] == null) {
                cur.trie[index] = new Trie();
            }
            cur = cur.trie[index];
        }
        cur.isEnd = true;
    }

    private int getLongestPrefix(Trie trie, String num) {
        Trie cur = trie;
        int count = 0;

        for (char c : num.toCharArray()) {
            int index = c - '0';
            
            if (cur.trie[index] == null) break;

            cur = cur.trie[index];
            count++;
        }

        return count;
    }
}
