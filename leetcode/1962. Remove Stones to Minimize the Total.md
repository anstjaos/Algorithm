class Solution {
    public int minStoneSum(int[] piles, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);

        for (int pile : piles) pq.offer(pile);

        for (int i = 0; i < k; i++) {
            int top = pq.poll();
            pq.offer(top - top / 2);
        }

        int result = 0;
        for (int val : pq) result += val;
        return result;
    }
}
