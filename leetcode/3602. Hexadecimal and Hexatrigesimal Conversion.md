class Solution {
    public String concatHex36(int n) {
        long n2 = (long) n * n;
        long n3 = (long) n * n * n;

        String hex = Long.toHexString(n2).toUpperCase();

        String base36 = "";
        while (n3 > 0) {
            long rem = n3 % 36;
            char ch;
            if (rem < 10) ch = (char) ('0' + rem);
            else ch = (char) ('A' + rem - 10);

            base36 = ch + base36;
            n3 /= 36;
        }

        return hex + base36;
    }
}
