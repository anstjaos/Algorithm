class Solution {
    public List<Integer> toggleLightBulbs(List<Integer> bulbs) {
        Set<Integer> set = new HashSet<>();

        for (int bulb : bulbs) {
            if (set.contains(bulb)) {
                set.remove(bulb);
            } else set.add(bulb);
        }

        List<Integer> result = new ArrayList<>(set);
        Collections.sort(result);

        return result;
    }
}
