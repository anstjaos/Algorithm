class TimeMap {
    class Data {
        String value;
        int timestamp;

        Data(String value, int timestamp) {
            this.value = value;
            this.timestamp = timestamp;
        }
    }

    private Map<String, List<Data>> map;

    public TimeMap() {
        map = new HashMap<>();
    }
    
    public void set(String key, String value, int timestamp) {
        List<Data> list = map.getOrDefault(key, new ArrayList<>());
        list.add(new Data(value, timestamp));
        map.put(key, list);
    }
    
    public String get(String key, int timestamp) {
        if (!map.containsKey(key)) {
            return "";
        }

        return binarySearch(map.get(key), timestamp);
    }

    private String binarySearch(List<Data> list, int timestamp) {
        int left = 0, right = list.size();

        while (left < right) {
            int mid = (left + right) / 2;

            if (list.get(mid).timestamp <= timestamp) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        return (left > 0) ? list.get(left - 1).value : "";
    }
}
