class Solution {
    public int firstMatchingIndex(String s) {
        int result = -1;
        for (int i = 0; i <= s.length() / 2; i++) {
            if (s.charAt(i) == s.charAt(s.length() - 1 - i)) {
                result = i;
                break;
            }
        }

        return result;
    }
}
