class Solution {
    public String getSmallestString(int n, int k) {
        char[] result = new char[n];
        k -= n;

        for (int i = n - 1; i >= 0 ; i--) {
            if (k > 25) {
                k -= 25;
                result[i] = 'z';
            } else {
                result[i] = (char)('a' + k);
                k = 0;
            }
        }
        return String.valueOf(result);
    }
}
