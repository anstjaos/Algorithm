/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<String>> printTree(TreeNode root) {
        List<List<String>> result = new ArrayList<>();
        if (root == null) return result;

        int height = getHeight(root);
        int m = height + 1;
        int n = (1 << m) - 1;

        String[][] temp = new String[m][n];
        for (int i = 0; i < m; i++) {
            Arrays.fill(temp[i], "");
        }

        dfs(root, 0, (n - 1) / 2, height, temp);

        for (int i = 0; i < m; i++) {
            List<String> row = new ArrayList<>();
            for (int j = 0; j < n; j++) {
                row.add(temp[i][j]);
            }

            result.add(row);
        }

        return result;
    }

    private int getHeight(TreeNode node) {
        if (node == null) return -1;
        return Math.max(getHeight(node.left), getHeight(node.right)) + 1;
    }

    private void dfs(TreeNode node, int row, int col, int height, String[][] matrix) {
        if (node == null) return;

        matrix[row][col] = Integer.toString(node.val);
        dfs(node.left, row + 1, col - (1 << (height - row - 1)), height, matrix);
        dfs(node.right, row + 1, col + (1 << (height - row - 1)), height, matrix);
    }
}
