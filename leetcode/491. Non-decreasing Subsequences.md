class Solution {
    public List<List<Integer>> findSubsequences(int[] nums) {
        Set<List<Integer>> result = new HashSet<>();

        getSubsequences(nums, result, 0, new ArrayList<>()); 
        return new ArrayList<>(result);
    }

    private void getSubsequences(int[] nums, Set<List<Integer>> result, int index, List<Integer> arr) {
        if (index >= nums.length) {
            if (arr.size() > 1) {
                result.add(new ArrayList<>(arr));
            }

            return;
        }

        if (arr.isEmpty() || nums[index] >= arr.get(arr.size() - 1)) {
            arr.add(nums[index]);
            getSubsequences(nums, result, index + 1, arr);
            arr.remove(arr.size() - 1);
        }

        getSubsequences(nums, result, index + 1, arr);
    }
}
