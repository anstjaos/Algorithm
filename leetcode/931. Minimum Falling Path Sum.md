class Solution {
    public int minFallingPathSum(int[][] matrix) {
        int row = matrix.length;
        int col = matrix[0].length;

        int[][] sum = new int[row][col];
        for (int i = 0; i < col; i++) {
            sum[row - 1][i] = matrix[row - 1][i];
        }

        for (int i = row - 2; i >= 0; i--) {
            for (int j = 0; j < col; j++) {
                int min = sum[i + 1][j];
                if (j - 1 >= 0) min = Math.min(min, sum[i + 1][j - 1]);
                if (j + 1 < col) min = Math.min(min, sum[i + 1][j + 1]);

                sum[i][j] = matrix[i][j] + min;
            }
        }

        int result = Integer.MAX_VALUE;
        for (int i = 0; i < col; i++) {
            result = Math.min(result, sum[0][i]);
        }

        return result;
    }
}
