class Solution {
    public int residuePrefixes(String s) {
        int distinctCount = 0;
        boolean[] isVisit = new boolean[26];

        int result = 0;
        for (int i = 0; i < s.length(); i++) {
            int c = s.charAt(i) - 'a';
            if (!isVisit[c]) {
                isVisit[c] = true;
                distinctCount++;
            }

            if (distinctCount == ((i + 1) % 3)) {
                result++;
            }
        }

        return result;
    }
}
