class Solution {
    public int[] vowelStrings(String[] words, int[][] queries) {
        int[] vowelCount = new int[words.length];
        for (int i = 0; i < words.length; i++) {
            String word = words[i];
            char first = word.charAt(0);
            char last = word.charAt(word.length() - 1);
            boolean isVowel = isVowel(first) && isVowel(last);

            vowelCount[i] = isVowel ? 1 : 0;
            if (i != 0) vowelCount[i] += vowelCount[i - 1];
        }

        int[] result = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int[] query = queries[i];
            if (query[0] == 0) {
                result[i] = vowelCount[query[1]];
            } else {
                result[i] = vowelCount[query[1]] - vowelCount[query[0] - 1];
            }
        }

        return result;
    }

    private boolean isVowel(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
}
