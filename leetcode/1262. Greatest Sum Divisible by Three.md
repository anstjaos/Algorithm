class Solution {
    public int maxSumDivThree(int[] nums) {
        int n = nums.length;

        int[][] dp = new int[n][3];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dp[i], 0);
        }

        dp[0][nums[0] % 3] = nums[0];
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < 3; j++) {
                dp[i][j] = dp[i - 1][j];
            }

            for (int j = 0; j < 3; j++) {
                int remain = (dp[i - 1][j] + nums[i]) % 3;
                dp[i][remain] = Math.max(dp[i][remain], dp[i - 1][j] + nums[i]);
            }
        }

        int result = 0;
        for (int i = 0; i < n; i++) {
            result = Math.max(result, dp[i][0]);
        }

        return result;
    }
}
