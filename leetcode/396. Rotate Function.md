class Solution {
    public int maxRotateFunction(int[] nums) {
        int[] dp = new int[nums.length];
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }

        for (int i = 0; i < nums.length; i++) {
            dp[0] += i * nums[i];
        }

        int result = dp[0];
        for (int i = 1; i < nums.length; i++) {
            dp[i] = dp[i - 1] + sum - nums.length * (nums[nums.length - i]);
            result = Math.max(result, dp[i]);
        }
        
        return result;
    }
}
