class Solution {
    public boolean checkDivisibility(int n) {
        int sum = 0, multi = 1;

        int temp = n;
        while (temp != 0) {
            int num = temp % 10;
            sum += num;
            multi *= num;
            temp /= 10;
        }

        return n % (sum + multi) == 0;
    }
}
