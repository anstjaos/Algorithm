class Solution {
    private int[] parent;
    private int[] rank;

    public int[] findRedundantConnection(int[][] edges) {
        int size = edges.length;
        parent = new int[size + 1];
        rank = new int[size + 1];

        for (int i = 1; i <= size; i++) {
            parent[i] = i;
        }

        for (int[] edge : edges) {
            int from = edge[0];
            int to = edge[1];

            if (find(from) == find(to)) {
                return edge;
            }
            union(from, to);
        }

        return new int[0];
    }

    private int find(int node) {
        if (parent[node] != node) {
            parent[node] = find(parent[node]);
        }

        return parent[node];
    }

    private void union(int u, int v) {
        int rootU = find(u);
        int rootV = find(v);

        if (rootU != rootV) {
            if (rank[rootU] > rank[rootV]) {
                parent[rootV] = rootU;
            } else if (rank[rootU] < rank[rootV]) {
                parent[rootU] = rootV;
            } else {
                parent[rootV] = rootU;
                rank[rootU]++;
            }
        }
    }
}
