class Solution {
    public long removeZeros(long n) {
        String val = String.valueOf(n);

        long result = 0;
        for (int i = 0; i < val.length(); i++) {
            int num = (int) (val.charAt(i) - '0');
            if (num == 0) continue;

            result = result * 10 + num;
        }

        return result;
    }
}
