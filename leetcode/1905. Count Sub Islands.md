class Solution {
    public int countSubIslands(int[][] grid1, int[][] grid2) {
        int result = 0;
        int row = grid1.length, col = grid1[0].length;

        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (grid2[i][j] == 0) continue;
                result += isSubIsland(grid1, grid2, i, j);
            }
        }

        return result;
    }

    private int isSubIsland(int[][] grid1, int[][] grid2, int x, int y) {
        int row = grid1.length, col = grid1[0].length;

        Queue<int[]> queue = new LinkedList<>();
        grid2[x][y] = 0;
        queue.add(new int[]{x, y});
        int result = grid1[x][y] != 0 ? 1 : 0;

        int[] dirX = {-1, 0, 1, 0}, dirY = {0, 1, 0, -1};

        while (!queue.isEmpty()) {
            int[] cur = queue.poll();

            for (int i = 0; i < 4; i++) {
                int nextX = cur[0] + dirX[i];
                int nextY = cur[1] + dirY[i];

                if (nextX < 0 || nextY < 0 || nextX >= row || nextY >= col) {
                    continue;
                }
                if (grid2[nextX][nextY] == 0) continue;
                if (grid1[nextX][nextY] == 0) {
                    result = 0;
                }

                grid2[nextX][nextY] = 0;
                queue.add(new int[]{nextX, nextY});
            }
        }

        return result;
    }
}
