class Solution {
    public String largestOddNumber(String num) {
        int lastIndex = -1;

        for (int i = num.length() - 1; i >= 0; i--) {
            if ((num.charAt(i) - '0') % 2 == 1) {
                lastIndex = i;
                break;
            }
        }

        return num.substring(0, lastIndex + 1);
    }
}
