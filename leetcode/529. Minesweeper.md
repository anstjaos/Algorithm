class Solution {
    class Point {
        int x, y;

        Point(int x, int y) { 
            this.x = x;
            this.y = y;
        }
    }

    int[] dirX = {-1, -1, 0, 1, 1, 1, 0, -1};
    int[] dirY = {0, 1, 1, 1, 0, -1, -1, -1};

    public char[][] updateBoard(char[][] board, int[] click) {
        int row = board.length, col = board[0].length;
        char[][] result = new char[row][col];
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                result[i][j] = board[i][j];
            }
        }

        if (board[click[0]][click[1]] == 'M') {
            result[click[0]][click[1]] = 'X';
            return result;
        }

        boolean[][] isVisit = new boolean[row][col];
        Queue<Point> queue = new LinkedList<>();
        queue.add(new Point(click[0], click[1]));

        while (!queue.isEmpty()) {
            Point cur = queue.poll();
            if (isVisit[cur.x][cur.y]) continue;
            isVisit[cur.x][cur.y] = true;

            int count = checkAdjust(cur.x, cur.y, board);
            if (count != 0) {
                result[cur.x][cur.y] = (char) ('0' + count);
                continue;
            }

            result[cur.x][cur.y] = 'B';

            for (int i = 0; i < 8; i++) {
                int nextX = cur.x + dirX[i];
                int nextY = cur.y + dirY[i];

                if (nextX < 0 || nextY < 0 || nextX >= board.length || nextY >= board[0].length) {
                    continue;
                }
                if (result[nextX][nextY] != 'E') continue;

                queue.add(new Point(nextX, nextY));
            }
        }

        return result;
    }

    private int checkAdjust(int x, int y, char[][] board) {
        int count = 0;

        for (int i = 0; i < 8; i++) {
            int nextX = x + dirX[i];
            int nextY = y + dirY[i];

            if (nextX < 0 || nextY < 0 || nextX >= board.length || nextY >= board[0].length) {
                continue;
            }
            if (board[nextX][nextY] == 'M') {
                count++;
            }
        }

        return count;
    }
}
