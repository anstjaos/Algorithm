class Solution {
    public int getLeastFrequentDigit(int n) {
        if (n == 0) return 0;

        int[] count = new int[10];
        while (n != 0) {
            count[n % 10]++;
            n /= 10;
        }

        int result = 0, minCount = Integer.MAX_VALUE;
        for (int i = 0; i < 10; i++) {
            if (count[i] == 0) continue;

            if (count[i] < minCount) {
                minCount = count[i];
                result = i;
            }
        }

        return result;
    }
}
