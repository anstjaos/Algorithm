class Solution {
    public int leastBricks(List<List<Integer>> wall) {
        Map<Integer, Integer> wallCount = new HashMap<>();

        for (List<Integer> list : wall) {
            int sum = 0;
            for (int i = 0; i < list.size() - 1; i++) {
                sum += list.get(i);
                wallCount.put(sum, wallCount.getOrDefault(sum, 0) + 1);
            }
        }

        int result = 0;
        for (int value : wallCount.values()) {
            result = Math.max(result, value);
        }

        return wall.size() - result;
    }
}
