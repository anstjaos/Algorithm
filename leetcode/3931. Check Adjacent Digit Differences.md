class Solution {
    public boolean isAdjacentDiffAtMostTwo(String s) {
        for (int i = 0; i < s.length() - 1; i++) {
            int cur = s.charAt(i) - '0';
            int next = s.charAt(i + 1) - '0';

            int diff = Math.abs(cur - next);
            if (diff > 2) {
                return false;
            }
        }

        return true;
    }
}
