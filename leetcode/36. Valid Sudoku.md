class Solution {
    public boolean isValidSudoku(char[][] board) {
        Set<String> visit = new HashSet<>();

        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                char number = board[i][j];
                if (number != '.') {
                    if (!visit.add(number + " in row " + i) 
                    || !visit.add(number + " in column " + j)
                    || !visit.add(number + " in block " + i / 3 + "-" + j / 3)) {
                        return false;
                    }
                }
            }
        }

        return true;
    }
}
