class Solution {
    public int lengthOfLongestSubstring(String s) {
        int[] index = new int[256];
        int left = 0, result = 0;
        Arrays.fill(index, -1);

        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (index[c] != -1 && index[c] >= left) {
                left = index[c] + 1;
            }
            result = Math.max(result, i - left + 1);
            index[c] = i;
        }

        return result;
    }
}
