class Solution {
    public int absDifference(int[] nums, int k) {
        Arrays.sort(nums);
        int max = 0, min = 0;

        for (int i = 0; i < k; i++) {
            min += nums[i];
            max += nums[nums.length - i - 1];
        }

        return Math.abs(max - min);
    }
}
