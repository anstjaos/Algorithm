class Solution {
    public String rearrangeString(String s, char x, char y) {
        StringBuilder sb = new StringBuilder();
        int count = 0;

        for (char c : s.toCharArray()) {
            if (c == x) {
                count++;
                continue;
            }
            sb.append(c);
        }

        for (int i = 0; i < count; i++) {
            sb.append(x);
        }

        return sb.toString();
    }
}
