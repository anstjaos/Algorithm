class Solution {
    public String mapWordWeights(String[] words, int[] weights) {
        int[] result = new int[words.length];

        for (int i = 0; i < words.length; i++) {
            String word = words[i];
            int sum = 0;
            for (char c : word.toCharArray()) {
                int idx = c - 'a';
                sum = (sum + weights[idx]) % 26;
            }

            result[i] = sum;
        }

        String word = "";
        for (int idx : result) {
            word += (char) ('a' + 25 - idx);
        }

        return word;
    }
}
