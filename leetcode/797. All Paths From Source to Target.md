class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> result = new ArrayList<>();
        List<Integer> list = new ArrayList<>();
        list.add(0);

        dfs(0, list, graph, graph.length - 1, result);
        return result; 
    }

    private void dfs(int src, List<Integer> list, int[][] graph, int dest, List<List<Integer>> result) {
        if (src == dest) {
            result.add(new ArrayList<>(list));
            return;
        }

        for (int n : graph[src]) {
            list.add(n);
            dfs(n, list, graph, dest, result);
            list.remove(list.size() - 1);
        }
    }
}
