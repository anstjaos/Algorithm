/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode bstToGst(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        postOrder(root, queue);
        
        int sum = 0;
        while (!queue.isEmpty()) {
            TreeNode cur = queue.poll();
            sum += cur.val;
            cur.val = sum;
        }

        return root;
    }

    private void postOrder(TreeNode cur, Queue<TreeNode> queue) {
        if (cur == null) return;

        postOrder(cur.right, queue);
        queue.add(cur);
        postOrder(cur.left, queue);
    }
}
