class Solution {
    public int scoreOfParentheses(String s) {
        int result = 0;
        Stack<Integer> stack = new Stack<>();

        for (char c : s.toCharArray()) {
            if (c == '(') {
                stack.push(result);
                result = 0;
            } else {
                result = stack.pop() + Math.max(2 * result, 1);
            }
        }

        return result;
    }
}
