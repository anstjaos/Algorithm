class Solution {
    public int closedIsland(int[][] grid) {
        int result = 0;
        int row = grid.length, col = grid[0].length;
        boolean[][] isVisit = new boolean[row][col];

        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (isVisit[i][j] || grid[i][j] == 1) continue;
                result += isIsland(grid, i, j, isVisit);
            }
        }

        return result;
    }

    private int isIsland(int[][] grid, int x, int y, boolean[][] isVisit) {
        int[] dirX = {-1, 0, 1, 0}, dirY = {0, 1, 0, -1};

        Queue<int[]> queue = new LinkedList<>();
        isVisit[x][y] = true;
        queue.add(new int[]{x, y});

        int result = 1;
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            
            for (int i = 0; i < 4; i++) {
                int nextX = cur[0] + dirX[i];
                int nextY = cur[1] + dirY[i];

                if (nextX < 0 || nextY < 0 || nextX >= grid.length || nextY >= grid[0].length) {
                    result = 0;
                    continue;
                }
                if (isVisit[nextX][nextY] || grid[nextX][nextY] == 1) continue;

                isVisit[nextX][nextY] = true;
                queue.add(new int[]{nextX, nextY});
            }
        }

        return result;
    }
}
