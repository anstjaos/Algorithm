class Solution {
    public int[] findDiagonalOrder(int[][] mat) {
        int row = mat.length, col = mat[0].length;
        int[] result = new int[row * col];

        int curRow = 0, curCol = 0;

        for (int i = 0; i < row * col; i++) {
            result[i] = mat[curRow][curCol];

            if ((curRow + curCol) % 2 == 0) {
                if (curCol == col - 1) curRow++;
                else if (curRow == 0) curCol++;
                else { 
                    curRow--; 
                    curCol++; 
                }
            } else {
                if (curRow == row - 1) curCol++;
                else if (curCol == 0) curRow++;
                else { 
                    curRow++; 
                    curCol--; 
                }
            }
        }

        return result;
    }
}
