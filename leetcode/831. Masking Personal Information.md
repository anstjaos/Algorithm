class Solution {
    public String maskPII(String s) {
        return s.contains("@") ? maskEmail(s) : maskPhoneNumber(s);
    }

    private String maskEmail(String s) {
        String[] splits = s.split("@");
        StringBuilder sb = new StringBuilder();

        sb.append(toLowerCase(splits[0].charAt(0)))
            .append("*****")
            .append(toLowerCase(splits[0].charAt(splits[0].length() - 1)));
        
        sb.append("@");

        sb.append(splits[1].toLowerCase());

        return sb.toString();
    }

    private char toLowerCase(char c) {
        if ('A' <= c && c <= 'Z') {
            return (char) (c - 'A' + 'a');
        }
        return c;
    }

    private String maskPhoneNumber(String s) {
        int numCount = 0;
        for (char c : s.toCharArray()) {
            if ('0' <= c && c <= '9') {
                numCount++;
            }
        }

        StringBuilder lastNumber = new StringBuilder();
        int count = 0;
        for (int i = s.length() - 1; i >= 0; i--) {
            char c = s.charAt(i);
            if ('0' <= c && c <= '9') {
                lastNumber.append(c);
                count++;
            }

            if (count == 4) break;
        }

        StringBuilder sb = new StringBuilder();
        switch (numCount) {
            case 10:
                sb.append("***-***-");
                break;
            case 11:
                sb.append("+*-***-***-");
                break;
            case 12:
                sb.append("+**-***-***-");
                break;
            default:
                sb.append("+***-***-***-");
                break;
        }

        sb.append(lastNumber.reverse());

        return sb.toString();
    }
}
