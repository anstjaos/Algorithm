class Solution {
    public List<Integer> findMissingElements(int[] nums) {
        Arrays.sort(nums);
        int min = nums[0], max = nums[nums.length - 1];
        
        List<Integer> result = new ArrayList<>();

        int cur = min, index = 0;
        while (cur < max) {
            if (index >= nums.length) {
                result.add(cur);
            } else if (nums[index] != cur) {
                result.add(cur);
            } else {
                index++;
            }
            cur++;
        }

        return result;
    }
}
