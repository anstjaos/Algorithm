class Solution {
    public int singleNumber(int[] nums) {
        Map<Integer, Long> map = Arrays.stream(nums).boxed()
            .collect(Collectors.groupingBy(v -> v, Collectors.counting()));

        int result = 0;
        for (Integer key : map.keySet()) {
            if (map.get(key) <= 1) {
                result = key.intValue();
                break;
            }
        }

        return result;
    }
}
