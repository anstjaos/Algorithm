class Solution {
    public boolean validDigit(int n, int x) {
        boolean isContain = false;
        
        while (n != 0) {
            int num = n % 10;
            if (x == num) {
                isContain = true;
            }
            if (n < 10 && n == x) {
                return false;
            }
            n /= 10;
        }

        return isContain;
    }
}
