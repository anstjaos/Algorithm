class Solution {
    public int numFactoredBinaryTrees(int[] arr) {
        final Integer DIV = 1000000007;

        Arrays.sort(arr);
        long result = 0;
        Map<Integer, Long> numCountMap = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            numCountMap.put(arr[i], numCountMap.getOrDefault(arr[i], 0L) + 1);

            for (int j = 0; j < i; j++) {
                if (arr[i] % arr[j] != 0) continue;

                int div = arr[i] / arr[j];
                if (!numCountMap.containsKey(div)) continue;

                long count = (numCountMap.get(div) * numCountMap.get(arr[j])) % DIV;
                numCountMap.put(arr[i], (numCountMap.get(arr[i]) + count) % DIV);
            }
            
            result += numCountMap.get(arr[i]);
            result %= DIV;
        }
        
        return (int) result;
    }
}
