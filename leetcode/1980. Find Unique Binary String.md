class Solution {
    public String findDifferentBinaryString(String[] nums) {
        int length = nums[0].length();
        Set<String> set = new HashSet<>(Arrays.asList(nums));

        return findBinaryString("", length, set);
    }

    private String findBinaryString(String cur, int length, Set<String> set) {
        if (cur.length() == length) {
            if (!set.contains(cur)) {
                return cur;
            }

            return "";
        }

        String temp1 = findBinaryString(cur + "0", length, set);
        if (!temp1.equals("")) {
            return temp1;
        }
        String temp2 = findBinaryString(cur + "1", length, set);
        if (!temp2.equals("")) {
            return temp2;
        }

        return "";
    }
}
