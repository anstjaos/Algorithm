/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> toDeleteSet = new HashSet<>();
        List<TreeNode> result = new ArrayList<>();
        for (int node : to_delete) {
            toDeleteSet.add(node);
        }

        removeNode(root, true, toDeleteSet, result);

        return result;
    }

    private TreeNode removeNode(TreeNode node, boolean isRoot, Set<Integer> toDeleteSet, List<TreeNode> result) {
        if (node == null) return null;

        boolean deleted = toDeleteSet.contains(node.val);
        if (isRoot && !deleted) {
            result.add(node);
        }

        node.left = removeNode(node.left, deleted, toDeleteSet, result);
        node.right = removeNode(node.right, deleted, toDeleteSet, result);

        return deleted ? null : node;
    }
}
