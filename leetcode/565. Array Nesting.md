class Solution {
    public int arrayNesting(int[] nums) {
        int result = 0;

        for (int num : nums) {
            if (num == -1) continue;
            int count = 0;

            while (nums[num] != -1) {
                count++;
                int prev = num;
                num = nums[num];
                nums[prev] = -1;
            }

            result = Math.max(result, count);
        }

        return result;
    }
}
