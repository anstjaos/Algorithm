class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> result = new ArrayList<>();

        int cur = 1;
        for (int i = 0; i < target.length; i++) {
            if (target[i] == cur) {
                result.add("Push");
                cur++;
            } else {
                while (target[i] != cur) {
                    result.add("Push");
                    result.add("Pop");
                    cur++;
                }
                i--;
            }
        }

        return result;
    }
}
