/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode bstFromPreorder(int[] preorder) {
        if (preorder == null) return null;
        return getBst(preorder, 0, preorder.length);
    }

    private TreeNode getBst(int[] preorder, int startIdx, int lastIdx) {
        if (startIdx >= lastIdx) return null;

        TreeNode root = new TreeNode(preorder[startIdx]);
        int rightIdx = startIdx + 1;
        while (rightIdx < lastIdx && preorder[rightIdx] < preorder[startIdx]) {
            rightIdx++;
        }

        root.left = getBst(preorder, startIdx + 1, rightIdx);
        root.right = getBst(preorder, rightIdx, lastIdx);

        return root;
    }
}
