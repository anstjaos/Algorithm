class Solution {
    public boolean scoreBalance(String s) {
        int[] scores = new int[s.length()];
        scores[0] = (int) s.charAt(0) - 'a' + 1;

        for (int i = 1; i < s.length(); i++) {
            scores[i] = scores[i - 1] + (int) (s.charAt(i) - 'a' + 1);
        }

        int sumAll = scores[s.length() - 1];
        for (int i = 0; i < s.length() - 1; i++) {
            int left = scores[i];
            int right = sumAll - scores[i];

            if (left == right) {
                return true;
            }
        }

        return false;
    }
}
