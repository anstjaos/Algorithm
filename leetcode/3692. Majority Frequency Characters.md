class Solution {
    public String majorityFrequencyGroup(String s) {
        int[] count = new int[26];
        for (char c : s.toCharArray()) {
            count[c - 'a']++;
        }

        Map<Integer, Integer> groupMap = new HashMap<>();
        for (int i = 0; i < 26; i++) {
            if (count[i] == 0) continue;

            groupMap.put(count[i], groupMap.getOrDefault(count[i], 0) + 1);
        }

        int max = 0, maxFrequency = 0;
        for (Map.Entry<Integer, Integer> entry : groupMap.entrySet()) {
            if (max == entry.getValue() && maxFrequency < entry.getKey()) {
                maxFrequency = entry.getKey();
            } else if (max < entry.getValue()) {
                max = entry.getValue();
                maxFrequency = entry.getKey();
            }
        }

        String result = "";
        for (int i = 0; i < 26; i++) {
            if (count[i] == maxFrequency) {
                result += (char) ('a' + i);
            }
        }

        return result;
    }
}
