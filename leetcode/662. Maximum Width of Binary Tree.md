/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    class Node {
        TreeNode node;
        int index;

        Node(TreeNode node, int index) {
            this.node = node;
            this.index = index;
        }
    }

    public int widthOfBinaryTree(TreeNode root) {
        if (root == null) return 0;

        Queue<Node> queue = new LinkedList<>();
        queue.add(new Node(root, 1));

        int result = 0;
        while (!queue.isEmpty()) {
            int first = queue.peek().index;
            int size = queue.size();
            int last = 0;

            for (int i = 0; i < size; i++) {
                Node cur = queue.poll();
                last = cur.index;

                if (cur.node.left != null) {
                    queue.add(new Node(cur.node.left, last * 2));
                }
                if (cur.node.right != null) {
                    queue.add(new Node(cur.node.right, last * 2 + 1));
                }
            }

            result = Math.max(result, last - first + 1);
        }

        return result;
    }
}
