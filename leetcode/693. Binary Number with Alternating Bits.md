class Solution {
    public boolean hasAlternatingBits(int n) {
        int before = -1;
        while (n != 0) {
            if (before == n % 2) return false;
            before = n % 2;
            n >>= 1;
        }
        return true;
    }
}

    public boolean hasAlternatingBits2(int n) {
    /*

        n ^ n>>1    1 1 1 1 1 1 1 1
