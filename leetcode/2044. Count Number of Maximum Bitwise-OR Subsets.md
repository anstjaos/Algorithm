class Solution {
    public int countMaxOrSubsets(int[] nums) {
        int maxOr = 0;
        for (int num : nums) {
            maxOr |= num;
        }

        return backtrack(nums, maxOr, 0, 0);
    }

    private int backtrack(int[] nums, int maxOr, int index, int cur) {
        if (index == nums.length) {
            return cur == maxOr ? 1 : 0;
        }

        if (cur == maxOr) {
            return 1 << (nums.length - index);
        }

        return backtrack(nums, maxOr, index + 1, cur | nums[index]) + 
            backtrack(nums, maxOr, index + 1, cur);
    }
}
