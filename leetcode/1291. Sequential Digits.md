class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        String nums = "123456789";
        int minLength = getLength(low), maxLength = getLength(high);

        List<Integer> result = new ArrayList<>();
        for (int i = minLength; i <= maxLength; i++) {
            for (int j = 0; j + i < 10; j++) {
                int num = Integer.parseInt(nums.substring(j, j + i));
                if (num > high) break;

                if (low <= num && num <= high) {
                    result.add(num);
                }
            }
        }

        return result;
    }

    private int getLength(int num) {
        int length = 0;
        while (num != 0) {
            length++;
            num /= 10;
        }

        return length;
    }
}
