/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    class MaxDepthNode {
        TreeNode node;
        int depth;

        MaxDepthNode(TreeNode node, int depth) {
            this.node = node;
            this.depth = depth;
        }
    }

    public TreeNode subtreeWithAllDeepest(TreeNode root) {
        MaxDepthNode result = new MaxDepthNode(root, 0);
        dfs(root, 0, result);

        return result.node;
    }

    private int dfs(TreeNode cur, int depth, MaxDepthNode maxDepthNode) {
        if (cur == null) {
            maxDepthNode.depth = Math.max(maxDepthNode.depth, depth);
            return depth;
        }

        int left = dfs(cur.left, depth + 1, maxDepthNode);
        int right = dfs(cur.right, depth + 1, maxDepthNode);

        if (left == right && left == maxDepthNode.depth) {
            maxDepthNode.node = cur;
        }

        return Math.max(left, right);
    }
}
