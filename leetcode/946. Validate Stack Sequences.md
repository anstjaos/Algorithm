class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        int pushedIndex = 0;
        int popIndex = 0;

        for (int val : pushed) {
            pushed[pushedIndex++] = val;
            while (pushedIndex > 0 && pushed[pushedIndex - 1] == popped[popIndex]) {
                pushedIndex--;
                popIndex++;
            }
        }

        return pushedIndex == 0;
    }
}
