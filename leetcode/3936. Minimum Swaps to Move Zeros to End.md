class Solution {
    public int minimumSwaps(int[] nums) {
        int count = 0;
        int left = 0, right = nums.length - 1;

        while (left < right) {
            while (left < nums.length && nums[left] != 0) left++;
            while (right >= 0 && nums[right] == 0) right--;

            if (left == nums.length || right == -1) break;
            if (left >= right) break;
            
            int temp = nums[left];
            nums[left] = nums[right];
            nums[right] = temp;
            count++;

            left++;
            right--;
        }

        return count;
    }
}
