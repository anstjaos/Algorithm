class Solution {
    class Trie {
        String word;
        Trie[] trie = new Trie[26];
    }

    public String longestWord(String[] words) {
        Trie trie = new Trie();

        for (String word : words) {
            insert(trie, word);
        }

        return dfs(trie, "");
    }

    private void insert(Trie trie, String word) {
        Trie cur = trie;
        for (char c : word.toCharArray()) {
            if (cur.trie[c - 'a'] == null) {
                cur.trie[c - 'a'] = new Trie();
            }
            cur = cur.trie[c - 'a'];
        }

        cur.word = word;
    }

    private String dfs(Trie trie, String result) {
        if (trie == null) return result;

        if (trie.word != null) {
            if (trie.word.length() > result.length()) {
                result = trie.word;
            }
        }

        for (Trie child : trie.trie) {
            if (child != null && child.word != null) {
                result = dfs(child, result);
            }
        }

        return result;
    }
}
