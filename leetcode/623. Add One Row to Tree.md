/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode addOneRow(TreeNode root, int val, int depth) {
        if (depth == 1) {
            TreeNode newRoot = new TreeNode(val);
            newRoot.left = root;
            return newRoot;
        }

        return addRow(root, val, depth, 1);
    }

    private TreeNode addRow(TreeNode cur, int val, int depth, int curDepth) {
        if (cur == null) return null;

        if (curDepth == depth - 1) {
            TreeNode left = new TreeNode(val);
            left.left = cur.left;
            cur.left = left;

            TreeNode right = new TreeNode(val);
            right.right = cur.right;
            cur.right = right;
            
            return cur;
        }

        cur.left = addRow(cur.left, val, depth, curDepth + 1);
        cur.right = addRow(cur.right, val, depth, curDepth + 1);
        return cur;
    }
}
