class Solution {
    public int[] countOppositeParity(int[] nums) {
        int[] result = new int[nums.length];
        int odd = 0, even = 0;

        for (int i = nums.length - 1; i >= 0; i--) {
            if ((nums[i] & 1) == 1) {
                result[i] = even;
                odd++;
            } else {
                result[i] = odd;
                even++;
            }
        }

        return result;
    }
}
