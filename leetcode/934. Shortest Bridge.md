class Solution {
    class Point {
        int x, y, count;

        Point(int x, int y, int count) {
            this.x = x;
            this.y = y;
            this.count = count;
        }
    }

    public int shortestBridge(int[][] grid) {
        int[] dirX = {-1, 0, 1, 0};
        int[] dirY = {0, 1, 0, -1};
        
        int row = grid.length, col = grid[0].length;

        Queue<Point> queue = new LinkedList<>();
        Point start = null;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (grid[i][j] == 1) {
                    start = new Point(i, j, 0);
                    break;
                }
            }
            if (start != null) break;
        }

        queue.add(start);
        grid[start.x][start.y] = 2;
        while (!queue.isEmpty()) {
            Point cur = queue.poll();

            for (int i = 0; i < 4; i++) {
                int nextX = cur.x + dirX[i];
                int nextY = cur.y + dirY[i];

                if (nextX < 0 || nextY < 0 || nextX >= row || nextY >= col) continue;
                if (grid[nextX][nextY] == 0 || grid[nextX][nextY] == 2) continue;

                grid[nextX][nextY] = 2;
                queue.add(new Point(nextX, nextY, 0));
            }
        }

        boolean[][] isVisit = new boolean[row][col];
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (grid[i][j] == 2) {
                    isVisit[i][j] = true;
                    queue.add(new Point(i, j, 0));
                }
            }
        }

        int result = 0;
        while (!queue.isEmpty()) {
            Point cur = queue.poll();

            for (int i = 0; i < 4; i++) {
                int nextX = cur.x + dirX[i];
                int nextY = cur.y + dirY[i];

                if (nextX < 0 || nextY < 0 || nextX >= row || nextY >= col) continue;
                if (isVisit[nextX][nextY]) continue;

                if (grid[nextX][nextY] == 1) {
                    result = cur.count;
                    break;
                }

                isVisit[nextX][nextY] = true;
                queue.add(new Point(nextX, nextY, cur.count + 1));
            }

            if (result != 0) break;
        }

        return result;
    }
}
