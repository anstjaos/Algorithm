class Solution {
    public int minSteps(int n) {
        if (n == 1) return 0;

        int result = 0;
        int length = 2;

        while (n > 1) {
            while (n % length == 0) {
                result += length;
                n /= length;
            }
            length++;
        }

        return result;
    }
}
