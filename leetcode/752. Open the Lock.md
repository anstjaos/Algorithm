class Solution {
    class Move {
        String num;
        int move;

        Move(String num, int move) {
            this.num = num;
            this.move = move;
        }
    }

    public int openLock(String[] deadends, String target) {
        Set<String> deadendSet = new HashSet<>(Arrays.asList(deadends));
        if (deadendSet.contains("0000")) return -1;

        Queue<Move> queue = new LinkedList<>();
        queue.offer(new Move("0000", 0));
        Set<String> visited = new HashSet<>();
        visited.add("0000");

        while (!queue.isEmpty()) {
            Move cur = queue.poll();
            String num = cur.num;
            int move = cur.move;

            if (num.equals(target)) {
                return move;
            }

            for (int i = 0; i < 4; i++) {
                for (int delta : new int[]{-1, 1}) {
                    int newDigit = (num.charAt(i) - '0' + delta + 10) % 10;
                    String newNum = num.substring(0, i) + newDigit + num.substring(i + 1);

                    if (!visited.contains(newNum) && !deadendSet.contains(newNum)) {
                        visited.add(newNum);
                        queue.add(new Move(newNum, move + 1));
                    }
                }
            }
        }

        return -1;
    }
}
