class Solution {
    public int[] concatWithReverse(int[] nums) {
        int[] result = new int[nums.length * 2];
        for (int i = 0; i < nums.length; i++) {
            result[i] = nums[i];
            result[nums.length + i] = nums[nums.length - i - 1];
        }

        return result;
    }
}
