class Solution {
    public int minOperations(String[] logs) {
        int depth = 0;
        for (String log : logs) {
            if (log.equals("../")) {
                if (depth == 0) continue;
                depth--;
            } else if (log.equals("./")) {
                continue;
            } else {
                depth++;
            }
        }

        return depth;
    }
}
