class Solution {
    public int[] constructArray(int n, int k) {
        int[] result = new int[n];
        for (int i = 0, a = 1, z = k + 1; i <= k; i++) {
            result[i] = i % 2 == 1 ? z-- : a++;
        }
        for (int i = k + 1; i < n;) {
            result[i] = ++i;
        }

        return result;
    }
}
