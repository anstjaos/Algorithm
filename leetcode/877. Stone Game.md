class Solution {
    public boolean stoneGame(int[] piles) {
        int alice = 0, bob = 0;
        boolean isAlice = true;
        int left = 0, right = piles.length - 1;

        for (int i = 0; i < piles.length; i++) {
            if (isAlice) {
                if (piles[left] >= piles[right]) {
                    alice += piles[left];
                    left++;
                } else {
                    alice += piles[right];
                    right--;
                }
            } else {
                if (piles[left] >= piles[right]) {
                    bob += piles[right];
                    right--;
                } else {
                    bob += piles[left];
                    left++;
                }
            }
            
            isAlice = !isAlice;
        }

        return alice >= bob;
    }
}
