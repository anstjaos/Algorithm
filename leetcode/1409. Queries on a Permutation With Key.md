class Solution {
    public int[] processQueries(int[] queries, int m) {
        List<Integer> arr = new ArrayList<>();
        int[] finalArr = new int[queries.length];
        for (int i = 1; i <= m; i++) {
            arr.add(i);
        }

        for (int i = 0; i < queries.length; i++) {
            int num = queries[i];

            if (arr.contains(num)) {
                finalArr[i] = arr.indexOf(num);
                arr.remove(arr.indexOf(num));
                arr.add(0, num);
            }
        }

        return finalArr;
    }
}
