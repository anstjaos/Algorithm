class Solution {
    public int findPairs(int[] nums, int k) {
        Arrays.sort(nums);

        int result = 0;
        for (int i = 0; i < nums.length - 1; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            result += binarySearch(nums, i, k);
        }

        return result;
    }

    private int binarySearch(int[] nums, int start, int k) {
        int left = start + 1, right = nums.length;

        while (left < right) {
            int mid = (left + right) / 2;

            if (nums[mid] - nums[start] == k) {
                return 1;
            } else if (nums[mid] - nums[start] > k) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }

        return 0;
    }
}
