class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int length = seq.length();
        int[] result = new int[length];
        int cur = 0;

        for (int i = 0; i < length; i++) {
            if (seq.charAt(i) == '(') {
                result[i] = cur++ % 2;
            } else {
                result[i] = --cur % 2;
            }
        }

        return result;
    }
}
