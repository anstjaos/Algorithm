class Solution {
    public int firstUniqueEven(int[] nums) {
        int[] count = new int[105];
        for (int num : nums) {
            count[num]++;
        }

        int min = -1;
        for (int num : nums) {
            if (num % 2 != 0) continue;

            if (count[num] == 1) {
                min = num;
                break;
            }
        }

        return min;
    }
}
