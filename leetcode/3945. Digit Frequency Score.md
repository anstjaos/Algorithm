class Solution {
    public int digitFrequencyScore(int n) {
        int[] count = new int[10];
        int temp = n;
        while (temp != 0) {
            count[temp % 10]++;
            temp /= 10;
        }

        int result = 0;
        for (int i = 1; i < 10; i++) {
            result += (i * count[i]);
        }

        return result;
    }
}
