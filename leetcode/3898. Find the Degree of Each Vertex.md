class Solution {
    public int[] findDegrees(int[][] matrix) {
        int[] degrees = new int[matrix.length];

        for (int i = 0; i < matrix.length; i++) {
            int count = 0;
            for (int j = 0; j < matrix.length; j++) {
                if (i == j) continue;
                
                count += matrix[i][j];
            }

            degrees[i] = count;
        }

        return degrees;
    }
}
