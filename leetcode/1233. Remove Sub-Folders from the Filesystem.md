class Solution {
    class Trie {
        Map<String, Trie> trieMap;
        boolean isEnd;

        Trie() {
            trieMap = new HashMap<>();
            isEnd = false;
        }
    }

    public List<String> removeSubfolders(String[] folders) {
        Trie dictionary = new Trie();
        Arrays.sort(folders);

        List<String> result = new ArrayList<>();

        for (String folder : folders) {
            String[] dicts = folder.substring(1).split("/");

            Trie cur = dictionary;
            boolean isSub = false;
            for (int i = 0; i < dicts.length; i++) {
                String dict = dicts[i];

                Trie next = cur.trieMap.getOrDefault(dict, new Trie());
                if (next.isEnd) {
                    isSub = true;
                }

                if (i == dicts.length - 1) {
                    next.isEnd = true;
                }
                cur.trieMap.put(dict, next);
                cur = next;
            }

            if (!isSub) {
                result.add(folder);
            }
        }

        return result;
    }
}
