class Solution {
    public int minFlipsMonoIncr(String s) {
        int oneCount = 0;
        int result = 0;

        for (char c : s.toCharArray()) {
            if (c == '1') oneCount++;
            else {
                result = Math.min(result + 1, oneCount);
            }
        }

        return result;
    }
}
