class Solution {
    public int wateringPlants(int[] plants, int capacity) {
        int steps = 0;
        int water = capacity, cur = -1;

        for (int i = 0; i < plants.length; i++) {
            if (water >= plants[i]) {
                steps += i - cur;
                water -= plants[i];
                cur = i;
            } else {
                steps += cur + i + 2;
                water = capacity - plants[i];
                cur = i;
            }
        }

        return steps;
    }
}
