class Solution {
    public List<Integer> pancakeSort(int[] arr) {
        int last = arr.length - 1;
        List<Integer> result = new ArrayList<>();

        while (last >= 0) {
            int max = Integer.MIN_VALUE;
            int index = -1;

            for (int i = 0; i <= last; i++) {
                if (arr[i] > max) {
                    max = arr[i];
                    index = i;
                }
            }

            if (index != 0) {
                result.add(index + 1);
                reverse(arr, index);
            }

            result.add(last + 1);
            reverse(arr, last);

            last--;
        }

        return result;
    }

    private void reverse(int[] arr, int k) {
        int left = 0, right = k;
        while (left < right) {
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        }
    }
}
