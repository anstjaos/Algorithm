class Solution {
    public int maxChunksToSorted(int[] arr) {
        int curSum = 0, expectSum = 0, chunck = 0;

        for (int i = 0; i < arr.length; i++) {
            curSum += arr[i];
            expectSum += i;

            if (curSum == expectSum) {
                chunck++;
            }
        }

        return chunck;
    }
}
