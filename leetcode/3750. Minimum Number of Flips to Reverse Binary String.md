class Solution {
    public int minimumFlips(int n) {
        List<Integer> list = new ArrayList<>();
        while (n != 0) {
            list.add(n % 2);
            n /= 2;
        }

        int length = list.size();
        int result = 0;

        for (int i = 0; i < length / 2; i++) {
            if (list.get(i) != list.get(length - 1 - i)) {
                result += 2;
            }
        }

        return result;
    }
}
