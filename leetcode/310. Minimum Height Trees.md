class Solution {
    public List<Integer> findMinHeightTrees(int n, int[][] edges) {
        if (n == 1) return List.of(0);

        List<Set<Integer>> adjust = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            adjust.add(new HashSet<>());
        }

        for (int[] edge : edges) {
            adjust.get(edge[0]).add(edge[1]);
            adjust.get(edge[1]).add(edge[0]);
        }

        List<Integer> leaves = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (adjust.get(i).size() == 1) leaves.add(i);
        }

        while (n > 2) {
            n -= leaves.size();
            List<Integer> temp = new ArrayList<>();
            for (int leave : leaves) {
                int j = adjust.get(leave).iterator().next();
                adjust.get(j).remove(leave);
                if (adjust.get(j).size() == 1) {
                    temp.add(j);
                }
            }
            leaves = temp;
        }

        return leaves;
    }
}
