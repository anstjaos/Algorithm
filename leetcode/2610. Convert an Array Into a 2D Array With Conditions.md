class Solution {
    public List<List<Integer>> findMatrix(int[] nums) {
        int[] count = new int[205];
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            count[num]++;
            set.add(num);
        }

        List<List<Integer>> result = new ArrayList<>();
        while (!set.isEmpty()) {
            int size = set.size();
            List<Integer> list = new ArrayList<>();
            List<Integer> deleteNum = new ArrayList<>();

            for (Integer num : set) {
                list.add(num);
                count[num]--;
                if (count[num] == 0) {
                    deleteNum.add(num);
                }
            }

            for (Integer num : deleteNum) {
                set.remove(num);
            }

            result.add(list);
        }

        return result;
    }
}
