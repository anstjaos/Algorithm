class Solution {
    public int[] findThePrefixCommonArray(int[] A, int[] B) {
        int length = A.length;
        boolean[] isExists = new boolean[length + 1];

        int[] result = new int[length];
        int count = 0;
        for (int i = 0; i < length; i++) {
            if (isExists[A[i]]) count++;
            else isExists[A[i]] = true;

            if (isExists[B[i]]) count++;
            else isExists[B[i]] = true;

            result[i] = count;
        }

        return result;
    }
}
