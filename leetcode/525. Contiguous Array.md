class Solution {
    public int findMaxLength(int[] nums) {
        int length = nums.length;
        Map<Integer, Integer> map = new HashMap<>();
        int sum = 0;
        int subLength = 0;

        for (int i = 0; i < length; i++) {
            sum += nums[i] == 0 ? -1 : 1;
            if (sum == 0) {
                subLength = i + 1;
            } else if (map.containsKey(sum)) {
                subLength = Math.max(subLength, i - map.get(sum));
            } else {
                map.put(sum, i);
            }
        }

        return subLength;
    }
}
