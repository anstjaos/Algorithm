class Solution {
    public int triangularSum(int[] nums) {
        int length = nums.length;
        while (length != 1) {
            for (int i = 1; i < length; i++) {
                nums[i - 1] = (nums[i] + nums[i - 1]) % 10;
            }
            length--;
        }

        return nums[0];
    }
}
